scriptTitle="Q50 Sensor Test";
scriptDescription="Buscando el fallo en Sensor";
scriptPackage="com.jacobo.q50.sensors";
scriptVersion=9;
scriptAuthor="Jacobo";

onInit = function() {
    quit = false;
    contador = 0.0;
};

main = function() {
    while (!quit) {
        contador = contador + 1.0;

        // PRUEBA 1: Intentar crear el sensor. 
        // Si falla aquí, el Introspector morirá con Sensor.createSensor
        try {
            miSensor = Sensor.createSensor("Prueba_Script", "P_SCR", "v");
            if (miSensor != null) {
                miSensor.setValue(contador);
                Log.logInfo("Sensor creado y actualizado");
            }
        } catch (e) {
            Log.logError("Fallo en Prueba 1: " + e);
        }

        // PRUEBA 2: Si la Prueba 1 falla, intentamos buscar un sensor 
        // que ya exista (pon el nombre de uno que tengas en Torque)
        // solo para ver si el objeto Sensor es accesible
        try {
            // Cambia "Revoluciones (Motor)" por el nombre exacto de un sensor real
            // o usa el ID si lo conoces.
            existente = Sensor.getSensorByName("Revoluciones (Motor)");
            if (existente != null) {
                Log.logInfo("Sensor existente encontrado: " + existente.getValue());
            }
        } catch (e) {
            Log.logError("Fallo en Prueba 2: " + e);
        }

        Time.sleep(2000);
    }
};

stop = function() {
    quit = true;
};
