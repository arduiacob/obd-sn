scriptTitle="Q50 Sniffer";
scriptDescription="Captura de bloque 2101";
scriptPackage="org.jacobo.q50.sniffer";
scriptVersion=4;
scriptAuthor="Jacobo";

onInit = function() {
    quit = false;
    debugLen = Sensor.createSensor("Q50_Len", "Len", "b");
    status = Sensor.createSensor("Q50_Status", "Status", "v");
    status.setValue(1);
};

main = function() {
    while (!quit) {
        // null para que no envíe header (o puedes poner "ATSH7E1" si lo necesitas)
        // Devuelve un String[]
        var responseArray = OBD.sendCommandGetResponse(null, "2101");
        
        if (responseArray != null) {
            var fullHex = "";
            
            // JEXL3 permite iterar arrays con esta sintaxis
            for (var line : responseArray) {
                fullHex = fullHex + line;
            }
            
            // Evaluamos la longitud de la cadena concatenada
            debugLen.setValue(fullHex.length()); 
            status.setValue(100);
            
            // Usamos el método correcto de la clase Log
            Log.logInfo("Q50_DATA: " + fullHex);
        } else {
            status.setValue(0);
            Log.logWarn("Q50_DATA: No hubo respuesta del adaptador");
        }
        
        Time.sleep(2000);
    }
};

stop = function() {
    quit = true;
};
