scriptTitle="Q50 Sniffer";
scriptDescription="Captura de bloque 2101";
scriptPackage="org.jacobo.q50.sniffer";
scriptVersion=2;
scriptAuthor="Jacobo";

onInit = function() {
    quit = false;
    // Usamos nombres cortos para evitar problemas de buffer
    debugLen = Sensor.createSensor("Q50_Len", "Len", "b");
    status = Sensor.createSensor("Q50_Status", "Status", "v");
    status.setValue(1);
};

main = function() {
    while (!quit) {
        // Usamos un bloque try-catch interno por si OBD.query falla
        try {
            var response = OBD.query("2101");
            
            if (response != null) {
                var len = response.length();
                debugLen.setValue(len);
                status.setValue(100);
                // Si esto falla, prueba con Log:log("...")
                Log.log("DATA:" + response);
            } else {
                status.setValue(0);
            }
        } catch (e) {
            status.setValue(-1);
        }

        Time.sleep(2000);
    }
};

stop = function() {
    quit = true;
};
